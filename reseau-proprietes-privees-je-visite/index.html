document.addEventListener("DOMContentLoaded", function() { // 1. On analyse l'adresse (URL) de la page pour trouver le paramètre "?mandat=" const urlParams = new URLSearchParams(window.location.search); const numeroMandat = urlParams.get('mandat'); // 2. Si un numéro de mandat est trouvé dans le lien, on l'injecte dans le formulaire if (numeroMandat) { const inputMandat = document.getElementById('champ-mandat'); if (inputMandat) { inputMandat.value = numeroMandat; } // Optionnel : On peut aussi l'afficher discrètement à l'écran pour rassurer l'acheteur const affichageMandat = document.getElementById('id-mandat-affiche'); if (affichageMandat) { affichageMandat.textContent = "Référence Mandat : " + numeroMandat; } } });
